<script type="text/javascript">var cityCode = function(data) {	$('#city').html(data.name);	$('#district').html(data.oun);};highCode = function(data){	$("#person").html(data.name);	$("#oun").html(data.oun);}$(document).ready(	function()	{		$('#autocompleteMe').Autocomplete(			{				source: '/tools/people2.lasso',				delay: 300,				helperClass: 'autocompleter',				selectClass: 'selectAutocompleter',				minchars: 2,				onSelect : cityCode,				onHighlight:highCode,				inputWidth:true			}		);	});</script>
