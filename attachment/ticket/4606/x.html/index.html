  <script>
    function Tab(index, name, id)
    {
        var obj = $("#" + name + index);
        $("#" + name + "s ~ div").hide();
        $("#" + name + "s > ul > li").removeClass("selected");
        $("#" + id + index).addClass("selected");    
        obj.show();    
    }
    function AdminTab(tab) {
        Tab(tab, 'AdminTab', 'adli'); 
    }
    $(document).ready(function() {
        AdminTab(0);
    });
    </script>
</head>
