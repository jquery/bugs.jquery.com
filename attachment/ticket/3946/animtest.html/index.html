Testing harness with datepicker and animations

<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01//EN"
     "http://www.w3.org/TR/html4/strict.dtd">
<html>
<head>
<meta http-equiv="Content-Type" content="text/html; charset=UTF-8">
<script type="text/javascript" src="jquery-1.2.6.js"></script>
<!--script type="text/javascript" src="jquery-1.3.js"></script-->
<script type="text/javascript" src="jquery-ui-effects-1.5.3.min.js"></script>
<!--script type="text/javascript" src="jquery-ui-effects-1.6rc5.min.js"></script-->
<script type="text/javascript" src="jquery.datepick.js"></script>
<script type="text/javascript">
$(function() {
	$('#animDatepicker').datepick({showOn: 'both'});

	$('#duration,#animation').change(function() {
		$('#animDatepicker').datepick('option',
			{showAnim: $('#animation').val(), duration: $('#duration').val()});
	});
});
</script>
<style type="text/css">
@import "jquery.datepick.css";
</style>
</head>
<body>
<p><span class="demoLabel">Show
	<select id="duration"><option value="slow">slow</option>
		<option value="normal" selected="selected">normal</option>
		<option value="fast">fast</option></select>
	<select id="animation"><option value="show">show</option>
		<option value="fadeIn">fadeIn</option>
		<option value="slideDown">slideDown</option>
		<option value="blind">blind</option>
		<option value="bounce">bounce</option>
		<option value="clip">clip</option>
		<option value="drop">drop</option>
		<option value="fold">fold</option>
		<option value="slide">slide</option></select>:</span>
	<input type="text" id="animDatepicker"/></p>
</body>
</html>
