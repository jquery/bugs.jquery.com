		//First hide all the subcategories

		$('img.arrow').click( 
			function() {
				if ($(this).attr("closed") == "1"){
					$(this).attr("src","images/arrow_open.gif");
					$(this).attr("closed","0");
					var decadeID = 'decade_' + $(this).attr('decadeID');
					var myUL = document.getElementById(decadeID);
					$(myUL).slideDown('slow');
					return false;
				}
				else{
					$(this).attr("src","images/arrow_closed.gif");
					$(this).attr("closed","1");
					var decadeID = 'decade_' + $(this).attr('decadeID');
					var myUL = document.getElementById(decadeID);
					$(myUL).slideUp('slow');
					return false;
				}
			});
		
		//show mouseover for arrow on hover	
		var imageName = "images/arrow_closed";
		$('span.arrowbox img').hover(function() {
			if ($(this).attr("closed") == "1"){
				imageName = "images/arrow_closed";
			}
			else{
				imageName = "images/arrow_open";
			}
			$(this).attr("src",imageName+"_on.gif");
			$(this).addClass('arrow-hover_IE');
			}, function() {
			if ($(this).attr("closed") == "1"){
				imageName = "images/arrow_closed";
			}
			else{
				imageName = "images/arrow_open";
			}
			$(this).attr("src",imageName+".gif");
			$(this).removeClass('arrow-hover_IE');
		});
