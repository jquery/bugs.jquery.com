var boxGlobal;

window.addEvent('domready', function() {
    boxGlobal = $('dialogBox');
    //Implement the following functions in every element
	Element.implement({
		show: function() {
            this.setStyle('display','block');
		},
		hide: function() {
			this.setStyle('display','none');
		}
	});
});


function showBox(){
    // boxGlobal.css('left', 300);
    boxGlobal.show();
}

function showFlicker(){
    // Does not flicker with MooTools. Flickers with jQuery.
    // http://mootools.net/docs/core/Element/Element.Style
    var width = boxGlobal.getStyle('width').toInt();
}

function closeBox(){
    boxGlobal.hide();
}
