<script type="text/javascript">

    jQuery.fn.saveDimensions = function() {
        return this.each(function() {
            $(this).data("outerWidth", $(this).outerWidth());
        });
    }

    var boxGlobalJQuery;

    $(function() {
        boxGlobalJQuery = $("#dialogBox");
        boxGlobalJQuery.saveDimensions();
    });

    function showBox() {
        boxGlobalJQuery.show();
    }

    function closeBox() {
        boxGlobalJQuery.saveDimensions().hide();
    }

    function getThem() {
        $("#values")
            .find("li:nth-child(1) span").empty().append(boxGlobalJQuery.css("width"))
            .end().find("li:nth-child(2) span").empty().append(boxGlobalJQuery.css("paddingLeft"))
            .end().find("li:nth-child(3) span").empty().append(boxGlobalJQuery.css("paddingRight"))
            .end().find("li:nth-child(4) span").empty().append(boxGlobalJQuery.css("borderLeftWidth"))
            .end().find("li:nth-child(5) span").empty().append(boxGlobalJQuery.css("borderRightWidth"))
            .end().find("li:nth-child(6) span").empty().append(boxGlobalJQuery.css("marginLeft"))
            .end().find("li:nth-child(7) span").empty().append(boxGlobalJQuery.css("marginRight"))
            .end().find("li:nth-child(8) span").empty().append(boxGlobalJQuery.css("left"))
            .end().find("li:nth-child(9) span").empty().append(boxGlobalJQuery.css("right"));
        
    }

    function getIt() {
        $("#val").empty().html(boxGlobalJQuery.data("outerWidth"));
    }

</script>
