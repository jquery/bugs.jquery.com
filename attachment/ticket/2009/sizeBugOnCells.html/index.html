		#overcell { display:block; position:absolute; background-color:orange; }
	</style>
	<script>
		$(function() {
			$('#overcell').css({opacity:0.5});
			$('body').mousemove(function(evt) {
				if(evt.target.tagName.toLowerCase() == 'td') { // move over a cell
					var $cell = $(evt.target), pos = $cell.offset();
					window.status = $('#overcell').css({display:'block', width:$cell.width(), height:$cell.height(), top:pos.top, left:pos.left }).attr('style');
					
				}
				else if(evt.target != $('#overcell')[0])
					$('#overcell').css({display:'none'})
			});
		});
	</script>
