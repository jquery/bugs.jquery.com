<script src="/fr/new/js/jquery-1.2.1.min.js"></script>
<script>
 jQuery(function($){
	divHeight = $("div").height();
	numberOfTries = 5;
	 $("#animator").one('click', function(){
		$("div").css({
			overflow: 'hidden',
			height: '0px',
			visibility: 'visible'
		}).animate(
			{height:divHeight},
			1500,
			(function() {
				alert(numberOfTries--);
				if(numberOfTries) {
					badFunc();
				}
			})
		);
	});
});
</script>
</head>
